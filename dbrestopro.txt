
SET statement_timeout = 0;
SET lock_timeout = 0;
SET client_encoding = 'UTF8';
SET standard_conforming_strings = on;
SET check_function_bodies = false;
SET client_min_messages = warning;
SET row_security = off;

CREATE EXTENSION IF NOT EXISTS plpgsql WITH SCHEMA pg_catalog;

COMMENT ON EXTENSION plpgsql IS 'PL/pgSQL procedural language';


SET search_path = public, pg_catalog;

SET default_tablespace = '';

SET default_with_oids = false;



CREATE TABLE akun (
    id character varying(10) NOT NULL,
    nama character varying(50),
    username character varying(15),
    password character varying(15),
    tipe character varying(10)
);


ALTER TABLE akun OWNER TO postgres;

CREATE TABLE akun_keuangan (
    kode_akun_keuangan character varying(15) NOT NULL,
    nama_akun_keuangan character varying(100),
    keterangan_akun_keuangan text
);


ALTER TABLE akun_keuangan OWNER TO postgres;


CREATE TABLE bahan (
    kode character varying(10) NOT NULL,
    nama character varying(50),
    jumlah double precision
);


ALTER TABLE bahan OWNER TO postgres;


CREATE TABLE catatan (
    kode character varying(15) NOT NULL,
    kode_perkiraan character varying(15),
    tanggal date,
    nama character varying(100),
    keterangan text,
    jumlah double precision,
    kode_akun_keuangan character varying(15),
    kode_user character varying(15)
);


ALTER TABLE catatan OWNER TO postgres;

CREATE TABLE kategori_meja (
    kode_kategori_meja character varying(10),
    nama_kategori_meja character varying(30)
);


ALTER TABLE kategori_meja OWNER TO postgres;

CREATE TABLE kategori_menu (
    kode_kategori_menu character varying(10),
    nama_kategori_menu character varying(30)
);


ALTER TABLE kategori_menu OWNER TO postgres;

CREATE TABLE meja (
    nama character varying(25),
    kategori character varying(15),
    status integer DEFAULT 0,
    kode character varying(10) NOT NULL
);


ALTER TABLE meja OWNER TO postgres;


CREATE TABLE menu (
    kode character varying(10) NOT NULL,
    nama character varying(100),
    kategori character varying(25),
    harga double precision,
    gambar character varying(20),
    kode_master_resep character varying(15)
);


ALTER TABLE menu OWNER TO postgres;

CREATE TABLE pembelian (
    tanggal date,
    kode_transaksi character varying(15) NOT NULL,
    kode_bahan character varying(15),
    nama character varying(50),
    jumlah double precision,
    harga double precision,
    kode_akun_keuangan character varying(15),
    kode_user character varying(15)
);


ALTER TABLE pembelian OWNER TO postgres;


CREATE TABLE perkiraan (
    kode_perkiraan character varying(15) NOT NULL,
    kategori_perkiraan character varying(15),
    nama_perkiraan character varying(100)
);


ALTER TABLE perkiraan OWNER TO postgres;

CREATE TABLE resep_detail (
    kode_master character varying(15),
    kode_bahan character varying(15),
    jumlah_pakai double precision,
    id integer NOT NULL
);


ALTER TABLE resep_detail OWNER TO postgres;

CREATE SEQUENCE resep_detail_id_seq
    START WITH 1
    INCREMENT BY 1
    NO MINVALUE
    NO MAXVALUE
    CACHE 1;


ALTER TABLE resep_detail_id_seq OWNER TO postgres;

ALTER SEQUENCE resep_detail_id_seq OWNED BY resep_detail.id;

CREATE TABLE resep_master (
    kode_resep character varying(15) NOT NULL,
    nama_resep character varying(100),
    status integer DEFAULT 0
);


ALTER TABLE resep_master OWNER TO postgres;

CREATE TABLE transaksi (
    kode_transaksi character varying(15),
    kode_menu character varying(15),
    harga_masing double precision,
    jumlah integer,
    tanggal timestamp without time zone DEFAULT now(),
    status integer DEFAULT 0,
    kode_user character varying(15),
    slot character varying(15),
    kode_meja character varying(15),
    tax double precision DEFAULT 0,
    statustax integer DEFAULT 0,
    kode_akun_keuangan character varying(15),
    diskon double precision DEFAULT 0,
    diskon_transaksi double precision DEFAULT 0,
    id integer NOT NULL
);


ALTER TABLE transaksi OWNER TO postgres;

CREATE SEQUENCE transaksi_id_seq
    START WITH 1
    INCREMENT BY 1
    NO MINVALUE
    NO MAXVALUE
    CACHE 1;


ALTER TABLE transaksi_id_seq OWNER TO postgres;

ALTER SEQUENCE transaksi_id_seq OWNED BY transaksi.id;

ALTER TABLE ONLY resep_detail ALTER COLUMN id SET DEFAULT nextval('resep_detail_id_seq'::regclass);

ALTER TABLE ONLY transaksi ALTER COLUMN id SET DEFAULT nextval('transaksi_id_seq'::regclass);
